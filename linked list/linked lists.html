Linked List Programs

Add at beginning

#include<stdio.h>
#include<malloc.h>

struct node{
	int data;
	struct node *next;
};

void insert_beg(struct node **start, int data);
struct node* create_node(int data);
void display(struct node *start);

int main(){
	struct node *start=NULL;
	int size, num, i;
	printf("\nADD AT BEGINNING\n\n");
	printf("Enter the number of elements to insert: ");
	scanf("%d", &size);
	for(i=0; i < size; i++){
		printf("Enter element %d: ", i+1);
		scanf("%d", &num);
		insert_beg(&start, num);
	}
	display(start);
	return 0;
}

void insert_beg(struct node **start, int data){
	struct node *temp; 
	if(*start == NULL){
		*start = create_node(data);
	}
	else{
		temp = create_node(data);
		temp->next = *start;
		*start = temp;
	}
}

struct node* create_node(int data){
	struct node *temp;
	temp = (struct node*) malloc(sizeof(struct node));
	temp->data = data;
	temp->next = NULL;
	return temp;
}

void display(struct node *start){
	struct node *temp = start;
	printf("\nThe linked list is: \n");
	printf("START->");
	while(temp != NULL){
		printf("%d -> ", temp->data);
		temp = temp->next;
	}
	printf("NULL");
}

		temp = temp->next;

		printf("Item not found\n");
		return;

	else if(temp1 == NULL && temp == *start){

		*start = temp1;
	}
	else{
		temp2 = create_node(num);
		temp2->next = temp;
		temp1->next = temp2;
	}
	
}
